# 1. Train the model (generates model_artifacts/)
python train.py

# 2. Run tests
pytest tests/test_api.py -v

# 3. Start locally (no Docker)
uvicorn app.main:app --reload --port 8000

# 4. Build & run with Docker Compose
docker-compose up --build -d

# 5. View MLflow UI
mlflow ui --backend-store-uri sqlite:///mlflow_tracking/mlflow.db
# → open http://localhost:5000

# 6. Test the API
curl -X POST http://localhost:8000/predict -H "Content-Type: application/json" -d '{"Pregnancies":2,"Glucose":120,"BloodPressure":72,"SkinThickness":20,"Insulin":80,"BMI":28.5,"DiabetesPedigreeFunction":0.35,"Age":33}'
